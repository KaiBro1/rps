def rpc():
 
  player = input('Rock (r), Paper (p) or Scissors (s)?') 
  
  print(player, 'vs', end =' ')
  
  chosen = randint(1,3)
  #print(chosen)
  
  if chosen == 1:
    computer = 'r'
  
  elif chosen == 2:
    computer = 'p'
    
  else:
    computer = 's'
    
  print(computer)
  
  if player == computer:
    print('DRAW!')
    rpc()
    
  if player == 'r' and computer == 's':
      print ('Player Wins!')
      rpc()
  
  if player == 'r' and computer == 'p':
      print ('Computer Wins!')
      rpc()
      
  if player == 's' and computer == 'p':
      print ('Player Wins!')
      rpc()
      
  if player == 's' and computer == 'r':
      print ('Computer Wins!')
      rpc()
      
  if player == 'p' and computer == 'r':
      print ('Player Wins!')
      rpc()
  
  if player == 'p' and computer == 's':
      print ('Computer Wins!')
      rpc()
  
  else:
     print ("Incorrect Input. Try (r) (p) (s)")
     rpc()
 
rpc()
